1 Code portability in Nim

Here's an benchmark example. 

 1: import timing, cpugpuarray, qexLite/metaUtils, math
 2: 
 3: proc test(vecLen, memLen: static[int]; N: int) =
 4:   var
 5:     x = newColorMatrixArray(vecLen,memLen,N) # array of N 3x3 single prec complex matrices
 6:     y = newColorMatrixArray(vecLen,memLen,N)
 7:     z = newColorMatrixArray(vecLen,memLen,N)
 8:     rep = 0                     # accumulates the number of runs
 9: 
10:   let
11:     mr = float(3 * 8 * x.T.N * x.T.N * N) / float(1024 * 1024 * 1024) # Resident memory in 2^30 bytes
12:     mt = 4 * mr / 3             # Memory transaction
13:     fp = float(8 * x.T.N * x.T.N * x.T.N * N) * 1e-9 # Floating point op / 10^9
14:   template timeit(label:string, s:untyped) =
15:     var
16:       R {.global.}:int
17:       T {.global.}:float
18:     threadSingle:
19:       R = 128                   # Base repeat
20:       T = 1.0                   # Time limit
21:     var t = timex(rep, R, s)    # Always warm up cache
22:     while true:
23:       threadSingle:
24:         R = min(64*R,max(R,int(R.float*0.8/t))) # set up to run for at least 0.8 sec or 64*R
25:       t = timex(rep, R, s)
26:       threadSingle: T -= t
27:       if T < 0: break
28:     threadSingle:               # Use the last R & t for performance measure
29:       printf("%8d %3d %d %-8s rep: %7d KB: %8.0f ms: %8.4f GF/s: %7.2f GB/s: %7.2f\n",
30:              N, vecLen, memLen, label, R, 1024*1024*mr, 1e3*t/R.float, fp*R.float/t, mt*R.float/t)
31: 
32:   threads:                      # CPU threads
33:     x := 0                      # set them to diagonal matrices on CPU
34:     y := 1
35:     z := 2
36:     timeit "CPU": x += y * z
37: 
38:   timeit "GPU5":                # includes kernel launching and synchronization
39:     onGpu(N, 32):               # Number of threads, threads per block
40:       x += y * z
41:   timeit "GPU6": onGpu(N, 64): x += y * z
42:   timeit "GPU7": onGpu(N, 128): x += y * z
43: 
44:   threads: timeit "CPU": x += y * z # back to CPU threads again
45: 
46:   let scale = 0.5 / (sqrt(3.0) * rep.float)
47:   threads:
48:     x *= scale
49:     var n = x.norm2
50:     threadSingle: echo "# Final scaled x.norm2: ",n,"  rep: ",rep
51:   x.free
52:   y.free
53:   z.free
54: 
55: for n in 8..26:
56:   staticFor v, 2, 7:
57:     when (1 shl v) >= (structsize(vectorizedElementType(float32)) div sizeof(float32)):
58:       staticFor ml, 1, 2:
59:         test(1 shl v, ml, 1 shl n)

The above can be compiled and run with 

nim cpp -d:SSE -d:AVX -d:CPUVLEN=256 -d:release ex2

2 Implementation details

The main container object in the example above is an array that can live on the CPU and also the GPU. This is defined as 

 1: when useGPU:
 2:   type
 3:     ArrayObj*[V,M:static[int],T] = object
 4:       p*: Coalesced[V,M,T]
 5:       n*: int
 6:       g*: GpuArrayObj[V,M,T]
 7:       lastOnGpu*: bool
 8:       unifiedMem*: bool
 9:       mem:pointer ## Pointer to the allocated memory.
10: else:
11:   type
12:     ArrayObj*[V,M:static[int],T] = object
13:       p*: Coalesced[V,M,T]
14:       n*: int
15:       mem:pointer ## Pointer to the allocated memory.
16: 
17: type
18:   GpuArrayObj*[V,M:static[int],T] = object
19:     p*: Coalesced[V,M,T]
20:     n*: int
21: 
22: type
23:   Coalesced*[V,M:static[int],T] = object
24:     ## `V`: Inner array length.
25:     ## `M`: Number of RegisterWords in a MemoryWord, the granularity of memory transactions.
26:     p*: ptr T                   ## pointer to an array of T
27:     n*: int                     ## the length of the array being coalesced
28:   CoalescedObj[V,M:static[int],T] = object
29:     o*: Coalesced[V,M,T]
30:     i*: int                     # the index to which we asks
31: 
32: template `[]`*(x:Coalesced, ix:int):untyped = CoalescedObj[x.V,x.M,x.T](o:x, i:ix)
33: template len*(x:Coalesced):untyped = x.n
34: 
35: template fromCoalesced*(x:CoalescedObj):untyped =
36:   const N = getSize(x.T) div (x.M*sizeof(RegisterWord))
37:   type A {.unchecked.}= ptr array[0,MemoryWord(x.M)]
38:   var r {.noinit.}: x.T
39:   let offset = (x.i div x.V)*N*x.V + x.i mod x.V
40:   staticfor j, 0, N-1: cast[A](r.addr)[j] = cast[A](x.o.p)[offset + j*x.V]
41:   r
42: 
43: type
44:   ShortVector*[V:static[int],E] = object
45:     a*:array[V,E]
46:   ShortVectorIndex* = distinct int
47:   VectorizedObj*[V,M:static[int],T] = object
48:     o*:Coalesced[V,M,T]
49:     i*:ShortVectorIndex
50: 
51: template `[]`*(x:Coalesced, ix:ShortVectorIndex):untyped = VectorizedObj[x.V,x.M,x.T](o:x,i:ix)
52: template veclen*(x:Coalesced):untyped = x.n div x.V

 2:   checkInit()

4 Offloading

 1: template cudaDefs(body: untyped): untyped {.dirty.} =
 2:   var gridDim{.global,importC,noDecl.}: CudaDim3
 3:   var blockIdx{.global,importC,noDecl.}: CudaDim3
 4:   var blockDim{.global,importC,noDecl.}: CudaDim3
 5:   var threadIdx{.global,importC,noDecl.}: CudaDim3
 6:   template getGridDim: untyped {.used.} = gridDim
 7:   template getBlockIdx: untyped {.used.} = blockIdx
 8:   template getBlockDim: untyped {.used.} = blockDim
 9:   template getThreadIdx: untyped {.used.} = threadIdx
10:   template getThreadNum: untyped {.used.} = blockDim.x * blockIdx.x + threadIdx.x
11:   template getNumThreads: untyped {.used.} = gridDim.x * blockDim.x
12:   bind inlineProcs
13:   inlineProcs:
14:     body
15: 
16: template cudaLaunch*(p: proc; blocksPerGrid,threadsPerBlock: SomeInteger;
17:                      arg: varargs[pointer,dataAddr]) =
18:   var pp: proc = p
19:   var gridDim, blockDim: CudaDim3
20:   gridDim.x = blocksPerGrid
21:   gridDim.y = 1
22:   gridDim.z = 1
23:   blockDim.x = threadsPerBlock
24:   blockDim.y = 1
25:   blockDim.z = 1
26:   var args: array[arg.len, pointer]
27:   for i in 0..<arg.len: args[i] = arg[i]
28:   #echo "really launching kernel"
29:   let err = cudaLaunchKernel(pp, gridDim, blockDim, addr args[0])
30:   if err:
31:     echo err
32:     quit cast[cint](err)
33: 
34: macro cuda*(s,p: untyped): auto =
35:   let ss = s.strVal
36:   p.expectKind nnkProcDef
37:   result = p
38:   result.addPragma parseExpr("{.codegenDecl:\""&ss&" $# $#$#\".}")[0]
39:   result.body = getAst(cudaDefs(result.body))
40:   var sl = newStmtList()
41:   sl.add( quote do:
42:     {.push checks: off.}
43:     {.push stacktrace: off.} )
44:   sl.add result
45:   result = sl
46: template cudaGlobal*(p: untyped): auto = cuda("__global__",p)

 1: template onGpu*(nn,tpb: untyped, body: untyped): untyped =
 2:   block:
 3:     var v = packVars(body, getGpuPtr)
 4:     type ByCopy {.bycopy.} [T] = object
 5:       d: T
 6:     proc kern(xx: ByCopy[type(v)]) {.cudaGlobal.} =
 7:       template deref(k: int): untyped = xx.d[k]
 8:       substVars(body, deref)
 9:     let ni = nn.int32
10:     let threadsPerBlock = tpb.int32
11:     let blocksPerGrid = (ni+threadsPerBlock-1) div threadsPerBlock
12:     cudaLaunch(kern, blocksPerGrid, threadsPerBlock, v)
13:     discard cudaDeviceSynchronize()
14: template onGpu*(nn: untyped, body: untyped): untyped = onGpu(nn, 64, body)
15: template onGpu*(body: untyped): untyped = onGpu(512*64, 64, body)

 1: proc kern(xx670162: ByCopy670160[type(v670158)])
 2:      {.codegenDecl: "__global__ $# $#$#".} =
 3:   var gridDim {.global, importC, noDecl.}: CudaDim3
 4:   var blockIdx {.global, importC, noDecl.}: CudaDim3
 5:   var blockDim {.global, importC, noDecl.}: CudaDim3
 6:   var threadIdx {.global, importC, noDecl.}: CudaDim3
 7:   template getGridDim(): untyped {.used.} = gridDim
 8:   template getBlockIdx(): untyped {.used.} = blockIdx
 9:   template getBlockDim(): untyped {.used.} = blockDim
10:   template getThreadIdx(): untyped {.used.} = threadIdx
11:   template getThreadNum(): untyped {.used.} = blockDim.x * blockIdx.x + threadIdx.x
12:   template getNumThreads(): untyped {.used.} = gridDim.x * blockDim.x
13:   inlineProcs:
14:     template deref(k670164: int): untyped =
15:       xx670162.d[k670164]
16:     substVars((x += y * z), deref)

4.2 Expression handling

 1: proc getVars*(v: var seq[NimNode], x,a: NimNode): NimNode =
 2:   proc recurse(it: NimNode, vars: var seq[NimNode], a: NimNode): NimNode =
 3:     var r0 = 0
 4:     var r1 = it.len - 1
 5:     case it.kind
 6:     of {nnkSym, nnkIdent}:
 7:       let i = vars.addIfNewSym(it)
 8:       if i>=0:
 9:         let ii = newLit(i)
10:         return newCall(a,ii)
11:     of nnkCallKinds: r0 = 1
12:     of nnkDotExpr: r1 = 0
13:     of {nnkVarSection,nnkLetSection}:
14:       result = it.cpNimNode
15:       for c in it:
16:         result.add c.cpNimNode
17:         for i in 0..(c.len-3):
18:           ignore.add c[i]
19:           result[^1].add c[i].cpNimNode
20:         result[^1].add c[^2].cpNimNode
21:         result[^1].add recurse(c[^1], vars, a)
22:       return
23:     else: discard
24:     result = it.cpNimNode
25:     for i in 0..<r0:
26:       result.add it[i].cpNimNode
27:     for i in r0..r1:
28:       result.add recurse(it[i], vars, a)
29:     for i in (r1+1)..<it.len:
30:       result.add it[i].cpNimNode
31:   ignore.newSeq(0)
32:   result = recurse(x, v, a)
33: 
34: macro packVarsStmt*(x: untyped, f: untyped): auto =
35:   var v = newSeq[NimNode](0)
36:   let a = ident("foo")
37:   let e = getVars(v, x, a)
38:   var p = newStmtList()
39:   for vs in v:
40:     p.add newCall(f,vs)
41:   result = p
42: 
43: macro packVars*(x: untyped, f: untyped): auto =
44:   var v = newSeq[NimNode](0)
45:   let a = ident("foo")
46:   let e = getVars(v, x, a)
47:   var p = newPar()
48:   if v.len==0:
49:     p.add newNimNode(nnkExprColonExpr).add(ident("Field0"),newLit(1))
50:   elif v.len==1:
51:     let vi = ident($v[0])
52:     p.add newNimNode(nnkExprColonExpr).add(ident("Field0"),newCall(f,vi))
53:   else:
54:     for vs in v:
55:       p.add newCall(f,vs)
56:   result = p
57: 
58: macro substVars*(x: untyped, a: untyped): auto =
59:   var v = newSeq[NimNode](0)
60:   let e = getVars(v, x, a)
61:   result = e

5 AST based overloading for array operations

 1: type ArrayIndex* = SomeInteger or ShortVectorIndex
 2: 
 3: template indexArray*(x: ArrayObj, i: ArrayIndex): untyped =
 4:   x.p[i]
 5: 
 6: macro indexArray*(x: ArrayObj{call}, y: ArrayIndex): untyped =
 7:   result = newCall(ident($x[0]))
 8:   for i in 1..<x.len:
 9:     let xi = x[i]
10:     result.add( quote do:
11:       indexArray(`xi`,`y`) )
12: 
13: template `[]`*(x: ArrayObj, i: ArrayIndex): untyped = indexArray(x, i)
